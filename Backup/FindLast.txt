
    ' Find the last picture in the selected folder
    Do
        If nIndex1 <= tPictureFiles.Count Then
            If tPictureFiles(nIndex1).FolderID <> lSelectedID Then
                If nIndex1 <> 1 Then nIndex1 = nIndex1 - 1
                Exit Do
            Else: nIndex1 = nIndex1 + 1
            End If
        Else
            If nIndex1 > tPictureFiles.Count Then nIndex1 = nIndex1 - 1
            Exit Do
        End If
    Loop
